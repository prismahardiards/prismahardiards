import numpy as np
import pandas as pd
from scipy.stats import f_oneway
import matplotlib.pyplot as plt

# Step 1:
group1 = [85, 90, 88, 75, 95]
group2 = [70, 65, 60, 75, 80]
group3 = [50, 55, 60, 65, 70]

# Step 2:
all_data = group1 + group2 + group3
grand_mean = np.mean(all_data)

# Step 3:
mean1, mean2, mean3 = np.mean(group1), np.mean(group2), np.mean(group3)

# Step 4:
n1, n2, n3 = len(group1), len(group2), len(group3)
ssb = n1 * (mean1 - grand_mean)**2 + n2 * (mean2 - grand_mean)**2 + n3 * (mean3 - grand_mean)**2

# Step 5:
ssw = sum((x - mean1)**2 for x in group1) + sum((x - mean2)**2 for x in group2) + sum((x - mean3)**2 for x in group3)

# Step 6:
df_between = 3 - 1
df_within = len(all_data) - 3

# Step 7:
msb = ssb / df_between
msw = ssw / df_within

# Step 8:
f_stat_manual = msb / msw

# Step 9:
f_stat_scipy, p_value = f_oneway(group1, group2, group3)

print("One-Way ANOVA Results")
print(f"Grand Mean: {grand_mean:.2f}")
print(f"SSB (Between-Group Variability): {ssb:.2f}")
print(f"SSW (Within-Group Variability): {ssw:.2f}")
print(f"MSB: {msb:.2f}, MSW: {msw:.2f}")
print(f"F-Statistic (Manual Calculation): {f_stat_manual:.2f}")
print(f"F-Statistic (SciPy): {f_stat_scipy:.2f}")
print(f"P-Value (SciPy): {p_value:.4f}")

# Step 10:
alpha = 0.05
if p_value < alpha:
    print("Result: Reject the null hypothesis. At least one group mean is significantly different.")
else:
    print("Result: Fail to reject the null hypothesis. No significant difference between group means.")

# Step 11:
data = pd.DataFrame({
    'Group': ['Group 1'] * len(group1) + ['Group 2'] * len(group2) + ['Group 3'] * len(group3),
    'Values': group1 + group2 + group3
})

plt.figure(figsize=(8, 6))
data.boxplot(by='Group', column=['Values'], grid=False, notch=True)
plt.title('Group Comparisons')
plt.suptitle("")
plt.xlabel('Groups')
plt.ylabel('Values')
plt.show()
