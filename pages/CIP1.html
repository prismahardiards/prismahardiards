import numpy as np
from scipy.stats import ttest_1samp, shapiro
import matplotlib.pyplot as plt

# Step 1:
data = [50, 52, 47, 49, 51, 53, 48, 50, 49, 52]  # Example sample data
population_mean = 50  # The known population mean

# Step 2:
alpha = 0.05
stat, p_normality = shapiro(data)
if p_normality > alpha:
    print("The data appears to be normally distributed.")
else:
    print("The data does not appear to be normally distributed. Be cautious with the t-test results.")

# Step 3:
mean = np.mean(data)

# Step 4:
std = np.std(data, ddof=1)

# Step 5:
n = len(data)

# Step 6:
t_stat_manual = (mean - population_mean) / (std / np.sqrt(n))

print(f"\nManually Calculated Statistics:")
print(f"Sample Mean: {mean:.2f}")
print(f"Sample Standard Deviation: {std:.2f}")
print(f"Sample Size: {n}")
print(f"T-Statistic (Manual Calculation): {t_stat_manual:.2f}")

# Step 7:
t_statistic, p_value = ttest_1samp(data, population_mean)

print("\nOne-Sample t-Test Results (Using SciPy)")
print(f"T-Statistic (SciPy): {t_statistic:.2f}")
print(f"P-Value: {p_value:.4f}")

# Step 8:
if p_value < alpha:
    print("Result: Reject the null hypothesis.")
    print("The sample mean is significantly different from the population mean.")
else:
    print("Result: Fail to reject the null hypothesis.")
    print("The sample mean is not significantly different from the population mean.")

# Step 9: _9_
plt.figure(figsize=(8, 5))
plt.hist(data, bins=5, alpha=0.7, label='Sample Data')
plt.axvline(x=population_mean, color='red', linestyle='--', label='Population Mean')
plt.axvline(x=mean, color='blue', linestyle='-', label='Sample Mean')
plt.xlabel('Value')
plt.ylabel('Frequency')
plt.title('Sample Data vs. Population Mean')
plt.legend()
plt.show()
