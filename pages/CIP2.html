import numpy as np
from scipy.stats import pearsonr
import matplotlib.pyplot as plt

# Step 1:
x = [43, 21, 25, 42, 57, 59, 79, 75, 87, 81]
y = [99, 65, 79, 75, 87, 81, 109, 105, 125, 119]

# Step 2:
r, p_value = pearsonr(x, y)
x_mean = np.mean(x)
y_mean = np.mean(y)

# Step 3:
numerator = sum((xi - x_mean) * (yi - y_mean) for xi, yi in zip(x, y))

# Step 4:
denominator = np.sqrt(
    sum((xi - x_mean)**2 for xi in x) * sum((yi - y_mean)**2 for yi in y)
)
# Step 5:
r_manual = numerator / denominator

print("Pearson Correlation Coefficient Results")
print(f"Correlation Coefficient (r): {r:.2f}")
print(f"P-Value: {p_value:.4f}")
print(f"Manually Calculated Correlation Coefficient (r): {r_manual:.2f}")

# Step 6:
plt.figure(figsize=(8, 5))
plt.scatter(x, y, alpha=0.7, label="Data Points")
plt.xlabel("Variable X")
plt.ylabel("Variable Y")
plt.title("Scatter Plot of X vs Y")
plt.axhline(y=np.mean(y), color='red', linestyle='--', label='Mean of Y')
plt.axvline(x=np.mean(x), color='blue', linestyle='--', label='Mean of X')
plt.legend()
plt.show()
